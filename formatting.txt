index=ldapsearch ObjectClass=user OR ObjectClass=computer
| where isnotnull(servicePrincipalName)
| mvexpand servicePrincipalName
| rename servicePrincipalName AS spn, sAMAccountName AS account, userAccountControl AS uac
| eval account_enabled = if(tonumber(uac) % 2 == 0, "yes", "no")
| table account, spn, account_enabled, distinguishedName


index=wineventlogs (EventCode=4768 OR EventCode=4769)
    Status=0x0
    earliest=-90d
| eval spn = coalesce(ServiceName, TargetUserName)
| eval spn = lower(trim(spn))
| stats max(_time) AS last_seen count AS ticket_count by spn, IpAddress
| eval last_seen_human = strftime(last_seen, "%Y-%m-%d %H:%M:%S")


| inputlookup ldap_spns.csv
| eval spn_lower = lower(trim(spn))

| join type=left spn_lower
    [search index=wineventlogs (EventCode=4768 OR EventCode=4769) Status=0x0 earliest=-90d
     | eval spn_lower = lower(trim(coalesce(ServiceName, TargetUserName)))
     | stats max(_time) AS last_seen count AS ticket_count by spn_lower]

| eval status = case(
    isnull(last_seen),          "dormant — never seen",
    last_seen < relative_time(now(), "-90d@d"), "dormant — 90+ days",
    1==1,                       "active")
| eval last_seen_human = if(isnull(last_seen), "never", strftime(last_seen, "%Y-%m-%d"))
| where status != "active"
| table account, spn, account_enabled, status, last_seen_human, ticket_count, distinguishedName
| sort - account_enabled, account

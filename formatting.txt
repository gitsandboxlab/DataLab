import os
import shutil
from pathlib import Path


def copy_repo_to_unc(repo_folder_name: str, unc_destination: str):
    """Copies all contents from C:\\Users\\<username>\\<repo_folder_name> to a UNC path.

    Args:
        repo_folder_name (str): The name of the folder inside your user directory
        unc_destination (str): The target UNC path (e.g., r'\\server\share\folder')
    """
    # 1. Dynamically get the current Windows user's profile path
    # This resolves to something like C:\Users\john_doe
    user_profile = Path(os.environ["USERPROFILE"])

    # 2. Build the full source path
    source_dir = user_profile / "Repo" / repo_folder_name

    # 3. Define the destination path
    dest_dir = Path(unc_destination)

    print(f"Source path: {source_dir}")
    print(f"Destination path: {dest_dir}")

    # Check if the source directory actually exists
    if not source_dir.exists():
        print(f"❌ Error: The source folder '{source_dir}' does not exist.")
        return

    try:
        # Create destination directories if they don't exist yet
        dest_dir.mkdir(parents=True, exist_ok=True)

        # 4. Copy the entire directory tree
        # dirs_exist_ok=True allows merging if the destination folder already exists
        shutil.copytree(source_dir, dest_dir, dirs_exist_ok=True)

        print("🚀 Successfully copied all files to the UNC directory!")

    except PermissionError:
        print(
            "❌ Permission Error: Ensure you have write permissions to the UNC path."
        )
    except Exception as e:
        print(f"❌ An error occurred: {e}")


if __name__ == "__main__":
    # --- CONFIGURE YOUR SETTINGS HERE ---
    # Example: If your project is at C:\Users\username\Repo\my-project
    MY_REPO_NAME = "my-project"

    # Example UNC Path (Note the 'r' prefix for raw string handling)
    UNC_TARGET = r"\\your-server-name\shared_folder\backup_destination"

    copy_repo_to_unc(MY_REPO_NAME, UNC_TARGET)

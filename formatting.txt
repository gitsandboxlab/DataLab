# Define target server and port
$SyslogServer = "192.168.1.100"  # Replace with your Linux box IP
$Port = 514

# Draft the RFC 3164 formatted syslog message
# <13> = Priority (Facility: user-level [1], Severity: Notice [5] -> (1*8)+5 = 13)
$Timestamp = (Get-Date).ToString("MMM dd HH:mm:ss")
$Hostname = $env:COMPUTERNAME
$Message = "<13>$Timestamp $Hostname TestApp: Hello rsyslog, this is a test packet from PowerShell!"

# Convert string message to bytes
$Bytes = [System.Text.Encoding]::ASCII.GetBytes($Message)

# Open UDP Socket and send data
$UdpClient = New-Object System.Net.Sockets.UdpClient
$UdpClient.Connect($SyslogServer, $Port)
[void]$UdpClient.Send($Bytes, $Bytes.Length)
$UdpClient.Close()

Write-Host "Sent fake UDP syslog packet to $SyslogServer:$Port" -ForegroundColor Green

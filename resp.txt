WITH ChurnDetected AS (
    -- Step 1: Look at the previous snapshot for each computer
    SELECT 
        SnapshotDate,
        ComputerName,
        LAG(SnapshotDate, 1) OVER (PARTITION BY ComputerName ORDER BY SnapshotDate) AS PrevSnapshotDate
    FROM DailyInventory
),
SessionFlags AS (
    -- Step 2: Mark the start of a new session (gap > 1 day or first time seen)
    SELECT 
        SnapshotDate,
        ComputerName,
        CASE 
            WHEN PrevSnapshotDate IS NULL OR SnapshotDate > DATEADD(day, 1, PrevSnapshotDate) THEN 1 
            ELSE 0 
        END AS IsNewSession
    FROM ChurnDetected
),
SessionGrouped AS (
    -- Step 3: Create a unique Session ID per computer using a running sum
    SELECT 
        SnapshotDate,
        ComputerName,
        SUM(IsNewSession) OVER (PARTITION BY ComputerName ORDER BY SnapshotDate) AS SessionID
    FROM SessionFlags
),
MaxSystemDate AS (
    -- Step 4: Find the most recent date available in the entire dataset
    SELECT MAX(SnapshotDate) AS MaxDate FROM DailyInventory
)
-- Step 5: Aggregate and apply conditional logic for active computers
SELECT 
    g.ComputerName,
    MIN(g.SnapshotDate) AS DateAdded,
    CASE 
        -- If the last day seen matches the ultimate max date in the data, it's still active
        WHEN MAX(g.SnapshotDate) = (SELECT MaxDate FROM MaxSystemDate) THEN 'Still Active'
        -- Otherwise, return the date it was last seen before it disappeared
        ELSE CONVERT(VARCHAR(10), MAX(g.SnapshotDate), 120) 
    END AS DateLastRemoved
FROM SessionGrouped g
GROUP BY 
    g.ComputerName, 
    g.SessionID
ORDER BY 
    g.ComputerName, 
    DateAdded;

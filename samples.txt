from prefect import get_run_logger, task

logger = logging.getLogger(**name**)

@task(name=“run_validation_checks”, retries=0)
def run_validation_checks(engine: Engine, checks: list[callable]) -> bool:
“””
Runs all validation checks. Returns False and logs failures if any check fails.
All checks run regardless so operator sees the full picture on failure.
“””
log = get_run_logger()
results = []

```
for check in checks:
    try:
        passed, msg = check(engine)
        results.append((passed, msg))
        level = "INFO" if passed else "WARNING"
        getattr(log, level.lower())(f"{'✓' if passed else '✗'} {msg}")
    except Exception as e:
        results.append((False, str(e)))
        log.error(f"✗ Validation check raised exception: {e}")

failed = [msg for passed, msg in results if not passed]

if failed:
    log.error(f"Validation failed — {len(failed)} check(s) did not pass. Model will not run.")
    return False

log.info(f"All {len(results)} validation check(s) passed.")
return True
```

@task(name=“execute_stored_procedure”, retries=1, retry_delay_seconds=30)
def execute_stored_procedure(engine: Engine, sp_name: str, timeout: int = 300):
“”“Executes a single stored procedure. Rolls back cleanly on error.”””
log = get_run_logger()
log.info(f”Executing: {sp_name}”)
try:
with engine.begin() as conn:
conn.execute(
text(f”EXEC {sp_name}”),
execution_options={“timeout”: timeout},
)
log.info(f”Completed: {sp_name}”)
except Exception as e:
log.error(f”Failed: {sp_name} — {e}”)
raise

def run_model(
engine: Engine,
sp_config: list[dict] | str,
validation_checks: list[callable],
) -> None:
“””
Validates then executes stored procedures in order.

```
Args:
    engine:             Your existing SQLAlchemy engine.
    sp_config:          Ordered list of SP dicts, or a JSON string.
                        Each entry: {"name": "usp_foo", "order": 1, "timeout": 300}
    validation_checks:  List of callables built via build_checks().
                        If empty, validation is skipped with a warning.
"""
log = get_run_logger()

# Accept raw JSON string (e.g. direct from Prefect Variable)
if isinstance(sp_config, str):
    sp_config = json.loads(sp_config)

# Validation gate
if not validation_checks:
    log.warning("No validation checks provided — skipping validation gate.")
    passed = True
else:
    passed = run_validation_checks(engine, validation_checks)

if not passed:
    log.warning("Skipping model build. Existing data model preserved.")
    return

# Execute SPs in order
ordered = sorted(sp_config, key=lambda x: x["order"])
log.info(f"Running {len(ordered)} stored procedure(s).")

for sp in ordered:
    execute_stored_procedure(
        engine,
        sp_name=sp["name"],
        timeout=sp.get("timeout", 300),
    )

log.info("Model build complete.")
```

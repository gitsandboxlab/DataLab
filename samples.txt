# samples
CREATE TABLE dbo.MailboxUsageStats (
    reportRefreshDate DATE,
    userPrincipalName NVARCHAR(255),
    displayName NVARCHAR(255),
    isDeleted BIT,
    deletedDate DATE,
    createdDate DATE,
    lastActivityDate DATE,
    itemCount INT,
    storageUsedInBytes BIGINT,
    deletedItemCount INT,
    deletedItemSizeInBytes BIGINT,
    deletedItemQuota BIGINT,
    hasArchive BIT,
    recipientType NVARCHAR(100),
    issueWarningQuotaInBytes BIGINT,
    prohitibSendQuotaInBytes BIGINT,
    prohibitSendReceivedQuotaInBytes BIGINT,
    reportPeriod INT,
    insertedDateTime DATETIME DEFAULT GETDATE() -- Helpful for tracking when the sync happened
);


CREATE PROCEDURE dbo.BulkInsertMailboxUsage
    @MailboxJson NVARCHAR(MAX)
AS
BEGIN
    SET NOCOUNT ON;

    INSERT INTO dbo.MailboxUsageStats (
        reportRefreshDate, userPrincipalName, displayName, isDeleted, 
        deletedDate, createdDate, lastActivityDate, itemCount, 
        storageUsedInBytes, deletedItemCount, deletedItemSizeInBytes, 
        deletedItemQuota, hasArchive, recipientType, issueWarningQuotaInBytes, 
        prohitibSendQuotaInBytes, prohibitSendReceivedQuotaInBytes, reportPeriod
    )
    SELECT *
    FROM OPENJSON(@MailboxJson)
    WITH (
        reportRefreshDate DATE,
        userPrincipalName NVARCHAR(255),
        displayName NVARCHAR(255),
        isDeleted BIT,
        deletedDate DATE,
        createdDate DATE,
        lastActivityDate DATE,
        itemCount INT,
        storageUsedInBytes BIGINT,
        deletedItemCount INT,
        deletedItemSizeInBytes BIGINT,
        deletedItemQuota BIGINT,
        hasArchive BIT,
        recipientType NVARCHAR(100),
        issueWarningQuotaInBytes BIGINT,
        prohitibSendQuotaInBytes BIGINT,
        prohibitSendReceivedQuotaInBytes BIGINT,
        reportPeriod INT
    );
END

import httpx
import os

# Use the exact URL you have in your Prefect config
url = "http://192.168.x.x:4200/api/health" 

print(f"Attempting to ping Prefect API at {url}...")

try:
    # We set a very strict 2-second timeout and NO retries
    response = httpx.get(url, timeout=2.0)
    print(f"✅ Success! Status Code: {response.status_code}")
except Exception as e:
    print(f"❌ Failed! Error: {e}")

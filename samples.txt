CREATE VIEW vw_M365UserReport AS
SELECT 
    u.UserID,
    u.DisplayName,
    u.UPN,
    u.IsActive,
    u.CreatedDate,
    -- Extract the Last Sign-In Date from JSON
    CAST(JSON_VALUE(u.SignInActivityJSON, '$.lastSignInDateTime') AS DATETIME2) AS LastSignInDate,
    
    -- Subquery to look up names from your LicenseMap table and comma-separate them
    (
        SELECT STRING_AGG(CAST(m.Product_Display_Name AS NVARCHAR(MAX)), ', ')
        FROM OPENJSON(u.LicensesJSON) 
        WITH (skuId UNIQUEIDENTIFIER) AS json_licenses
        LEFT JOIN LicenseMap m ON json_licenses.skuId = m.GUID
    ) AS FriendlyLicenseNames
FROM M365Users u;

CREATE TABLE M365Users (
    UserID UNIQUEIDENTIFIER PRIMARY KEY,
    DisplayName NVARCHAR(255),
    UPN NVARCHAR(255),
    IsActive BIT,
    CreatedDate DATETIME2,
    SignInActivityJSON NVARCHAR(MAX), -- Full JSON object
    LicensesJSON NVARCHAR(MAX)        -- Full JSON array
);

CREATE PROCEDURE sp_BulkLoadM365Users
    @jsonInput NVARCHAR(MAX)
AS
BEGIN
    SET NOCOUNT ON;

    -- Wipe the existing data
    TRUNCATE TABLE M365Users;

    -- Insert the new batch
    INSERT INTO M365Users (UserID, DisplayName, UPN, IsActive, CreatedDate, SignInActivityJSON, LicensesJSON)
    SELECT 
        UserID, 
        DisplayName, 
        UPN, 
        IsActive, 
        CreatedDate, 
        SignInActivityJSON, 
        LicensesJSON
    FROM OPENJSON(@jsonInput)
    WITH (
        UserID UNIQUEIDENTIFIER,
        DisplayName NVARCHAR(255),
        UPN NVARCHAR(255),
        IsActive BIT,
        CreatedDate DATETIME2,
        SignInActivityJSON NVARCHAR(MAX) AS JSON,
        LicensesJSON NVARCHAR(MAX) AS JSON
    );
END

CREATE TABLE [dbo].[ManagedDevices_Inventory] (
    [id] [nvarchar](100) PRIMARY KEY,
    [deviceName] [nvarchar](255),
    [userPrincipalName] [nvarchar](255),
    [userDisplayName] [nvarchar](255),
    [lastSyncDateTime] [datetimeoffset](7),
    [complianceState] [nvarchar](50),
    [operatingSystem] [nvarchar](100),
    [osVersion] [nvarchar](100),
    [serialNumber] [nvarchar](100),
    [model] [nvarchar](100),
    [manufacturer] [nvarchar](100),
    [enrolledDateTime] [datetimeoffset](7),
    [totalStorageSpaceInBytes] [bigint],
    [freeStorageSpaceInBytes] [bigint],
    [easDeviceId] [nvarchar](100),
    [LastUpdated] [datetime] DEFAULT GETDATE()
);

CREATE PROCEDURE [dbo].[sp_BulkLoadManagedDevices_Fresh]
    @DeviceJson NVARCHAR(MAX)
AS
BEGIN
    SET NOCOUNT ON;

    -- 1. Wipe the table
    TRUNCATE TABLE [dbo].[ManagedDevices_Inventory];

    -- 2. Bulk Insert from JSON
    INSERT INTO [dbo].[ManagedDevices_Inventory] (
        id, deviceName, userPrincipalName, userDisplayName, 
        lastSyncDateTime, complianceState, operatingSystem, 
        osVersion, serialNumber, model, manufacturer, 
        enrolledDateTime, totalStorageSpaceInBytes, 
        freeStorageSpaceInBytes, easDeviceId
    )
    SELECT 
        id, deviceName, userPrincipalName, userDisplayName, 
        lastSyncDateTime, complianceState, operatingSystem, 
        osVersion, serialNumber, model, manufacturer, 
        enrolledDateTime, totalStorageSpaceInBytes, 
        freeStorageSpaceInBytes, easDeviceId
    FROM OPENJSON(@DeviceJson)
    WITH (
        id nvarchar(100),
        deviceName nvarchar(255),
        userPrincipalName nvarchar(255),
        userDisplayName nvarchar(255),
        lastSyncDateTime datetimeoffset(7),
        complianceState nvarchar(50),
        operatingSystem nvarchar(100),
        osVersion nvarchar(100),
        serialNumber nvarchar(100),
        model nvarchar(100),
        manufacturer nvarchar(100),
        enrolledDateTime datetimeoffset(7),
        totalStorageSpaceInBytes bigint,
        freeStorageSpaceInBytes bigint,
        easDeviceId nvarchar(100)
    );
END

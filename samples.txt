from prefect.blocks.webhook import Webhook
from prefect.utilities.asyncutils import run_coro_as_sync

def call_hook(block_name: str, payload: dict):
    """
    The actual Transport execution.
    Handles the async Prefect block inside a sync wrapper.
    """
    async def _execute_call():
        # Load the block and emit the POST request
        webhook_block = await Webhook.load(block_name)
        return await webhook_block.call(payload=payload)

    # Bridge the gap: run the async code and return the result
    return run_coro_as_sync(_execute_call())


from prefect.variables import Variable
from prefect.utilities.asyncutils import run_coro_as_sync

def update_variable_key(variable_name: str, key: str, value: any):
    """
    Updates a single key inside a Prefect JSON variable without 
    disturbing other existing keys.
    """
    # 1. Get the current dictionary
    # We default to an empty dict if it doesn't exist
    current_var = Variable.get(variable_name, default={})
    
    # Handle both Variable objects and raw dicts
    data = current_var.value if hasattr(current_var, 'value') else current_var
    
    if not isinstance(data, dict):
        raise ValueError(f"Variable '{variable_name}' must be a JSON/dict to update a key.")

    # 2. Update the specific key
    data[key] = value

    # 3. Save the whole object back
    run_coro_as_sync(
        Variable.set(
            name=variable_name,
            value=data,
            overwrite=True
        )
    )
    return data

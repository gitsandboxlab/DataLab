# 1. Force the Worker to give up quickly if the server lags
prefect config set PREFECT_CLIENT_MAX_RETRIES=0
prefect config set PREFECT_API_REQUEST_TIMEOUT=5.0

# 2. Slow down the 'Panic' frequency of the worker
$env:PREFECT_WORKER_HEARTBEAT_SECONDS=30
$env:PREFECT_CLIENT_RETRY_EXTRA_BACKOFF=5

# 3. Disable the 'High-Speed' logging that floods the terminal
prefect config set PREFECT_LOGGING_LEVEL=ERROR

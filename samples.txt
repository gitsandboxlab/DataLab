CREATE PROCEDURE [dbo].[usp_BulkInsertGroupMembers]
    @GroupName NVARCHAR(255),
    @JsonData NVARCHAR(MAX),
    @Status NVARCHAR(50)
AS
BEGIN
    SET NOCOUNT ON;

    IF @JsonData IS NOT NULL AND @JsonData <> '[]' AND @JsonData <> ''
    BEGIN
        INSERT INTO [dbo].[GroupMembershipAudit] (GroupName, UserObjectID, MembershipStatus)
        SELECT 
            @GroupName, 
            [id], 
            @Status
        FROM OPENJSON(@JsonData)
        WITH (
            [id] NVARCHAR(100) -- For Active, Graph returns 'id'. For PIM, we will map it below.
        );
    END
END

SELECT 
    u.UserID,
    u.DisplayName,
    (
        SELECT STRING_AGG(m.Product_Display_Name, ', ')
        FROM (
            SELECT DISTINCT m.Product_Display_Name
            FROM OPENJSON(u.LicensesJSON) 
            WITH (
                skuId UNIQUEIDENTIFIER '$.skuId' -- This extracts the ID from each object
            ) AS j
            INNER JOIN LicenseMap m ON j.skuId = m.GUID -- Joins to your CSV lookup table
            WHERE m.Product_Display_Name IS NOT NULL
        ) AS UniqueLicenses
    ) AS FriendlyLicenseNames
FROM M365Users u;

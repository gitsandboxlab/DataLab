SELECT 
    u.UserID,
    u.DisplayName,
    u.UPN,
    -- The Subquery starts here
    (
        SELECT STRING_AGG(CAST(m.Product_Display_Name AS NVARCHAR(MAX)), ', ')
        FROM (
            SELECT DISTINCT m.Product_Display_Name
            FROM OPENJSON(u.LicensesJSON) 
            WITH (skuId UNIQUEIDENTIFIER '$.skuId') AS j
            LEFT JOIN LicenseMap m ON j.skuId = m.GUID
            WHERE m.Product_Display_Name IS NOT NULL
        ) AS m -- This 'm' is local to this subquery only
    ) AS FriendlyLicenseNames -- This is the alias for the whole result
FROM M365Users u;

import os
import requests
from requests.auth import HTTPBasicAuth
from dotenv import load_dotenv

load_dotenv()

# Config
DOMAIN = os.getenv("JIRA_DOMAIN") 
CLOUD_ID = os.getenv("CLOUD_ID") # Get from /_edge/tenant_info
EMAIL = os.getenv("JIRA_EMAIL")
API_TOKEN = os.getenv("JIRA_API_TOKEN")
ISSUE_KEY = "PROJ-123" 

auth = HTTPBasicAuth(EMAIL, API_TOKEN)
headers = {
    "Accept": "application/json",
    "Content-Type": "application/json"
}

def get_jsm_form_data():
    # --- STEP 1: Get all forms attached to the issue directly ---
    # This replaces the 'proforma.forms' property call
    list_forms_url = f"https://api.atlassian.com/jsm/forms/v1/cloud/{CLOUD_ID}/issue/{ISSUE_KEY}/form"
    
    response = requests.get(list_forms_url, auth=auth, headers=headers)
    
    if response.status_code != 200:
        print(f"Error fetching forms list: {response.status_code}")
        print(response.text)
        return

    forms = response.json() # This returns a list of form objects
    
    if not forms:
        print(f"No forms found on issue {ISSUE_KEY}")
        return

    # Loop through each form found on the issue
    for form_info in forms:
        form_id = form_info['uuid'] # The new API uses 'uuid' instead of 'id'
        print(f"Processing Form: {form_info.get('name')} (ID: {form_id})")

        # --- STEP 2: Get Answers for this specific Form ---
        answers_url = f"https://api.atlassian.com/jsm/forms/v1/cloud/{CLOUD_ID}/issue/{ISSUE_KEY}/form/{form_id}/answers"
        ans_response = requests.get(answers_url, auth=auth, headers=headers)
        
        if ans_response.status_code == 200:
            ans_data = ans_response.json()
            print(f"Answers: {ans_data.get('answers')}")
            # Here is where you would call your SQL Insert function
        else:
            print(f"Could not get answers for form {form_id}: {ans_response.status_code}")

if __name__ == "__main__":
    get_jsm_form_data()

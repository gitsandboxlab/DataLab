CREATE PROCEDURE [dbo].[sp_BulkInsertIntuneSettings]
    @JsonData NVARCHAR(MAX)
AS
BEGIN
    SET NOCOUNT ON;

    INSERT INTO [dbo].[IntuneComplianceReport] (
        IntuneSettingId, Setting, SettingName, DeviceId, DeviceName, 
        UserId, UserEmail, UserName, UserPrincipalName, DeviceModel, 
        [ComplianceState], ComplianceGracePeriodExpiration
    )
    SELECT 
        id, 
        setting, 
        settingName, 
        deviceId, 
        deviceName, 
        userId, 
        userEmail, 
        userName, 
        userPrincipalName, 
        deviceModel, 
        [state],
        -- Convert UTC to Pacific Standard Time
        CAST(complianceGracePeriodExpirationDateTime AT TIME ZONE 'UTC' AT TIME ZONE 'Pacific Standard Time' AS DATETIME2)
    FROM OPENJSON(@JsonData)
    WITH (
        id NVARCHAR(255),
        setting NVARCHAR(MAX),
        settingName NVARCHAR(255),
        deviceId UNIQUEIDENTIFIER,
        deviceName NVARCHAR(255),
        userId NVARCHAR(255),
        userEmail NVARCHAR(255),
        userName NVARCHAR(255),
        userPrincipalName NVARCHAR(255),
        deviceModel NVARCHAR(255),
        [state] NVARCHAR(50),
        complianceGracePeriodExpirationDateTime DATETIMEOFFSET -- Read raw UTC first
    );
END
